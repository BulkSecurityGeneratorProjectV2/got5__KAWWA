052/**
053 * A component used to render a recursive tree structure, with
054 * expandable/collapsable/selectable nodes. The data that is displayed by the
055 * component is provided as a {@link TreeModel}. A secondary model, the
056 * {@link TreeExpansionModel}, is used to track which nodes have been expanded.
057 * The optional {@link TreeSelectionModel} is used to track node selections (as
058 * currently implemented, only leaf nodes may be selected).
059 * <p/>
060 * The Tree component uses special tricks to support recursive rendering of the
061 * Tree as necessary.
062 * 
063 * @since 5.3
064 * @tapestrydoc
065 */
066@SuppressWarnings({ "rawtypes", "unchecked", "unused" })
067@Events({ TreeConstants.NODE_SELECTED, TreeConstants.NODE_UNSELECTED })
068@Import(stylesheet = { "classpath:/net/atos/kawwaportal/components/assets/tree/tree.css" })
069public class Tree {
070
071        @Inject
072        @Symbol(value = JQuerySymbolConstants.SUPPRESS_PROTOTYPE)
073        private boolean prototype;
074        /**
075         * The model that drives the tree, determining top level nodes and making
076         * revealing the overall structure of the tree.
077         */
078        @Parameter(required = true, autoconnect = true)
079        private TreeModel model;
080
081        /**
082         * Allows the container to specify additional CSS class names for the outer
083         * DIV element. The outer DIV always has the class name "t-tree-container";
084         * the additional class names are typically used to apply a specific size
085         * and width to the component.
086         */
087        @Parameter(name = "class", defaultPrefix = BindingConstants.LITERAL)
088        private String className;
089
090        /**
091         * Optional parameter used to inform the container about what TreeNode is
092         * currently rendering; this is primarily used when the label parameter is
093         * bound.
094         */
095        @Property
096        @Parameter
097        private TreeNode node;
098
099        /**
100         * Used to control the Tree's expansion model. By default, a persistent
101         * field inside the Tree component stores a
102         * {@link DefaultTreeExpansionModel}. This parameter may be bound when more
103         * control over the implementation of the expansion model, or how it is
104         * stored, is required.
105         */
106        @Parameter(allowNull = false, value = "defaultTreeExpansionModel")
107        private TreeExpansionModel expansionModel;
108
109        /**
110         * Used to control the Tree's selections. When this parameter is bound, then
111         * the client-side Tree will track what is selected or not selected, and
112         * communicate this (via Ajax requests) up to the server, where it will be
113         * recorded into the model. On the client-side, the Tree component will add
114         * or remove the {@code t-selected-leaf-node-label} CSS class from
115         * {@code span.t-tree-label} for the node.
116         */
117        @Parameter
118        private TreeSelectionModel selectionModel;
119
120        /**
121         * Optional parameter used to inform the container about the value of the
122         * currently rendering TreeNode; this is often preferable to the TreeNode,
123         * and like the node parameter, is primarily used when the label parameter
124         * it bound.
125         */
126        @Parameter
127        private Object value;
128
129        /**
130         * A renderable (usually a {@link Block}) that can render the label for a
131         * tree node. This will be invoked after the {@link #value} parameter has
132         * been updated.
133         */
134        @Property
135        @Parameter(value = "block:defaultRenderTreeNodeLabel")
136        private RenderCommand label;

150        private static RenderCommand RENDER_CLOSE_TAG = new RenderCommand() {
151                public void render(MarkupWriter writer, RenderQueue queue) {
152                        writer.end();

158                if (prototype)
159                        path = "/net/atos/kawwaportal/components/assets/tree/tree-jquery.js";

164        private static RenderCommand RENDER_LABEL_SPAN = new RenderCommand() {
165                public void render(MarkupWriter writer, RenderQueue queue) {
166                        // writer.element("span", "class", "t-tree-label");
167                        writer.element("a", "class", "t-tree-label", "href", "#");
168                }
169        };

171        /**
172         * Renders a single node (which may be the last within its containing node).
173         * This is a mix of immediate rendering, and queuing up various Blocks and
174         * Render commands to do the rest. May recursively render child nodes of the
175         * active node.
176         * 
177         * @param node
178         *            to render
179         * @param isLast
180         *            if true, add "t-last" attribute to the LI element
181         * @return command to render the node
182         */
183        private RenderCommand toRenderCommand(final TreeNode node,
184                        final boolean isLast) {
185                return new RenderCommand() {
186                        public void render(MarkupWriter writer, RenderQueue queue) {
187                                // Inform the component's container about what value is being
188                                // rendered
189                                // (this may be necessary to generate the correct label for the
190                                // node).
191                                Tree.this.node = node;
192
193                                value = node.getValue();
194
195                                writer.element("li");
196
197                                if (isLast)
198                                        writer.attributes("class", "t-last");
199
200                                Element e = writer.element("span", "class", "t-tree-icon");
201
202                                if (node.isLeaf())
203                                        e.addClassName("t-leaf-node");
204                                else if (!node.getHasChildren())
205                                        e.addClassName("t-empty-node");
206
207                                boolean hasChildren = !node.isLeaf() && node.getHasChildren();
208                                boolean expanded = hasChildren
209                                                && expansionModel.isExpanded(node);
210
211                                String clientId = jss.allocateClientId(resources);
212
213                                JSONObject spec = new JSONObject("clientId", clientId);
214
215                                e.attribute("id", clientId);
216
217                                spec.put("leaf", node.isLeaf());
218
219                                if (hasChildren) {
220                                        Link expandChildren = resources.createEventLink(
221                                                        "expandChildren", node.getId());
222                                        Link markExpanded = resources.createEventLink(
223                                                        "markExpanded", node.getId());
224                                        Link markCollapsed = resources.createEventLink(
225                                                        "markCollapsed", node.getId());
226
227                                        spec.put("expandChildrenURL", expandChildren.toString())
228                                                        .put("markExpandedURL", markExpanded.toString())
229                                                        .put("markCollapsedURL", markCollapsed.toString());
230
231                                        if (expanded)
232                                                spec.put("expanded", true);
233                                } else {
234                                        if (selectionModel != null) {
235                                                // May need to address this in the future; in other tree
236                                                // implementations I've constructed,
237                                                // folders are selectable, and selections even propagate
238                                                // up and down the tree.
239
240                                                Link selectLeaf = resources.createEventLink("select",
241                                                                node.getId());
242
243                                                spec.put("selectURL", selectLeaf.toString());
244                                                if (selectionModel.isSelected(node)) {
245                                                        spec.put("selected", true);
246                                                }
247                                        }
248                                }
249
250                                jss.addInitializerCall("treeNode", spec);
251
252                                writer.end(); // span.tx-tree-icon
253
254                                // From here on in, we're pushing things onto the queue.
255                                // Remember that
256                                // execution order is reversed from order commands are pushed.
257
258                                queue.push(RENDER_CLOSE_TAG); // li
259
260                                if (expanded)
261
262                                {
263                                        queue.push(new RenderNodes(node.getChildren()));
264                                }
265
266                                queue.push(label);
267                        }
268                }
269
270                ;
271        }
272
273        /**
274         * Renders an &lt;ul&gt; element and renders each node recursively inside
275         * the element.
276         */
277        private class RenderNodes implements RenderCommand {
278                private final Flow<TreeNode> nodes;
279
280                public RenderNodes(List<TreeNode> nodes) {
281                        assert !nodes.isEmpty();
282
283                        this.nodes = F.flow(nodes).reverse();
284                }
285
286                public void render(MarkupWriter writer, final RenderQueue queue) {
287                        writer.element("ul");
288                        queue.push(RENDER_CLOSE_TAG);
289
290                        queue.push(toRenderCommand(nodes.first(), true));
291
292                        nodes.rest().each(new Worker<TreeNode>() {
293                                public void work(TreeNode element) {
294                                        queue.push(toRenderCommand(element, false));
295                                }
296                        });
297                }
298
299        }
300
301        public String getContainerClass() {
302                return className == null ? "t-tree-container" : "t-tree-container "
303                                + className;
304        }
305
306        Object onExpandChildren(String nodeId) {
307                TreeNode container = model.getById(nodeId);
308
309                expansionModel.markExpanded(container);
310
311                return new RenderNodes(container.getChildren());
312        }
313
314        Object onMarkExpanded(String nodeId) {
315                expansionModel.markExpanded(model.getById(nodeId));
316
317                return new JSONObject();
318        }
319
320        Object onMarkCollapsed(String nodeId) {
321                expansionModel.markCollapsed(model.getById(nodeId));
322
323                return new JSONObject();
324        }
325
326        Object onSelect(String nodeId,
327                        @RequestParameter("t:selected") boolean selected) {
328                TreeNode node = model.getById(nodeId);
329
330                String event;
331
332                if (selected) {
333                        selectionModel.select(node);
334
335                        event = TreeConstants.NODE_SELECTED;
336                } else {
337                        selectionModel.unselect(node);
338
339                        event = TreeConstants.NODE_UNSELECTED;
340                }
341
342                CaptureResultCallback<Object> callback = CaptureResultCallback.create();
343
344                resources.triggerEvent(event, new Object[] { nodeId }, callback);
345
346                final Object result = callback.getResult();
347
348                if (result != null)
349                        return result;
350
351                return new JSONObject();
352        }
353
354        public TreeExpansionModel getDefaultTreeExpansionModel() {
355                if (defaultTreeExpansionModel == null)
356                        defaultTreeExpansionModel = new DefaultTreeExpansionModel();
357
358                return defaultTreeExpansionModel;
359        }
360
361        /**
362         * Returns the actual {@link TreeExpansionModel} in use for this Tree
363         * component, as per the expansionModel parameter. This is often, but not
364         * always, the same as {@link #getDefaultTreeExpansionModel()}.
365         */
366        public TreeExpansionModel getExpansionModel() {
367                return expansionModel;
368        }
369
370        /**
371         * Returns the actual {@link TreeSelectionModel} in use for this Tree
372         * component, as per the {@link #selectionModel} parameter.
373         */
374        public TreeSelectionModel getSelectionModel() {
375                return selectionModel;
376        }
377
378        public Object getRenderRootNodes() {
379                return new RenderNodes(model.getRootNodes());
380        }
381
382        /**
383         * Clears the tree's {@link TreeExpansionModel}.
384         */
385        public void clearExpansions() {
386                expansionModel.clear();
387        }

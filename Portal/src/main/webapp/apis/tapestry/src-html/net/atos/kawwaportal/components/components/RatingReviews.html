001package net.atos.kawwaportal.components.components;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import net.atos.kawwaportal.components.data.ratings.CriteriaDefinition;
007import net.atos.kawwaportal.components.data.ratings.RatingCriteria;
008import net.atos.kawwaportal.components.data.ratings.Review;
009
010import org.apache.tapestry5.annotations.Import;
011import org.apache.tapestry5.annotations.Parameter;
012import org.apache.tapestry5.annotations.Property;
013
014/**
015 * <p>
016 *              This Component is used to provide a list of reviews.
017 * 
018 *              <br/><br/>
019 * 
020 *      To use it, call the component with '&lt;div t:type="kawwa2/RatingReviews" t:id="ratingReviews"
021                        t:globalRating="globalRating" t:ratingReviews="ratingReviews"
022                        t:criteriasDefinition="criteriasDefinition"/&gt;'.
023 * </p>
024 * 
025 * <p>
026 *              There is parameters to call for the displayingform:
027 *              <ul>
028 *                      <li>t:globalRating="&lt;&lt;!&gt;&gt;" ==> Replace &lt;&lt;!&gt;&gt; by the object of the average value for the main rating criteria of the reviews.</li>
029 *                      <li>t:ratingReviews="&lt;&lt;!&gt;&gt;" ==> This parameter is Mandatory. Replace &lt;&lt;!&gt;&gt; by the list of reviews you want to be displayed.</li>
030 *                      <li>t:criteriasDefinition="&lt;&lt;!&gt;&gt;" ==> This parameter is Mandatory. Replace &lt;&lt;!&gt;&gt; by the list of CriteriaDefinition that will define the criterias passed with the instance of ratingReviews.</li>
031 *              </ul>
032 * </p>
033 * 
034 * @tapestrydoc
035 */
036
037@Import(stylesheet = "classpath:/net/atos/kawwaportal/components/assets/ratings/ratings.css")
038public class RatingReviews {
039        
040        @Parameter
041        @Property
042        private RatingCriteria globalRating;
043
044        @Parameter(required=true)
045        @Property
046        @SuppressWarnings("unused")
047        private ArrayList<Review> ratingReviews;
048        
049        @Parameter(required=true)
050        @Property
051        private List<CriteriaDefinition> criteriasDefinition;
052        
053        @Property
054        @SuppressWarnings("unused")
055        private Review currentRating;
056        
057        @Property
058        @SuppressWarnings("unused")
059        private RatingCriteria currentCriteria;
060        
061        public boolean getGlobalRatingIsSetted() {
062                
063                return globalRating != null;
064        }
065        
066        public CriteriaDefinition getCriteriaDefinitionByRatingCriteria(RatingCriteria ratingCriteria) {
067        
068        CriteriaDefinition criteriaDefinition = null;
069        
070        for(CriteriaDefinition currentCriteria : criteriasDefinition) {
071                
072                if(ratingCriteria.getRef().equals(currentCriteria.getRef()))
073                        criteriaDefinition = currentCriteria;
074        }
075        
076        return criteriaDefinition;
077    }
078}

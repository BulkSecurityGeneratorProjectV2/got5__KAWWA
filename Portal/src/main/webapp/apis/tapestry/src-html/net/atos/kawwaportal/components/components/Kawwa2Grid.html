205        static class CachingDataSource implements GridDataSource {
206
207                private final GridDataSource delegate;
208
209                private boolean availableRowsCached;
210
211                private int availableRows;
212
213                CachingDataSource(GridDataSource delegate) {
214                        this.delegate = delegate;
215                }
216
217                public int getAvailableRows() {
218                        if (!availableRowsCached) {
219                                availableRows = delegate.getAvailableRows();
220                                availableRowsCached = true;
221                        }
222
223                        return availableRows;
224                }
225
226                public void prepare(int startIndex, int endIndex,
227                                List<SortConstraint> sortConstraints) {
228                        delegate.prepare(startIndex, endIndex, sortConstraints);
229                }
230
231                public Object getRowValue(int index) {
232                        return delegate.getRowValue(index);
233                }
234
235                @SuppressWarnings("unchecked")
236                public Class getRowType() {
237                        return delegate.getRowType();
238                }
239        }
240
241        void setupDataSource() {
242                this.cachedSource = new CachingDataSource(this.source);
243                this.availableRows = cachedSource.getAvailableRows();
244        }
245
246        @SetupRender
247        public void initSource(MarkupWriter writer) {
248                setupDataSource();
249
250                // We add a Listener for adding the caption element and the summary
251                // attribute to the table
252                writer.addListener(new MarkupWriterListener() {
253
254                        public void elementDidStart(Element element) {
255                                if (element.getName().equalsIgnoreCase("table")) {
256
257                                        if (cr.isBound("information")) {
258                                                element.forceAttributes("summary",
259                                                                information.getTableSummary());
260                                                Element caption = element.element("caption");
261                                                caption.text(information.getTableCaption());
262                                        }
263                                }
264
265                        }
266
267                        public void elementDidEnd(Element element) {
268                        }
269                });
270
271                if (clientId != null && !"".equals(clientId)) {
272                        uniqueId = support.allocateClientId(clientId);
273
274                } else {
275                        uniqueId = support.allocateClientId(resources);
276
277                }
278
279                this.rowPerPageZone = "rowPerPageZone_" + getClientId();
280
281                try {
282                        if (cookieEnabled) {
283
284                                this.rowsPerPage = Integer.parseInt(cookies
285                                                .readCookieValue(resources.getCompleteId()));
286                        }
287                } catch (NumberFormatException nbfEx) {
288
289                        // Set the number of default row per page to display.
290                        // Value would be taken during first time rendering of Kawwagrid.
291                        if (rowsPerPage == null) {
292
293                                rowsPerPage = defaultRowPerPage;
294                        }
295                }
296        }
297
298        public Integer getRowsPerPage() {
299
300                return this.rowsPerPage;
301        }
302
303        public void setRowsPerPage(Integer rowsPerPage) {
304
305                this.rowsPerPage = rowsPerPage;
306
307                // Writing the value to a cookie
308                if (cookieEnabled) {
309                        cookies.writeCookieValue(resources.getCompleteId(),
310                                        this.rowsPerPage + "", maxAge);
311                }
312
313                // trigger event when select drop down value changed
314
315                resources
316                                .triggerEvent(
317                                                KawwaEventsConstants.GRID_EVENT_ROW_PER_PAGE_SELECT,
318                                                null, null);
319
320        }

346        /**
347         * Event handler for Rows Per page
348         * 
349         * @return Object
350         */
351        @OnEvent(KawwaEventsConstants.SELECT_ROWPERPAGE)
352        public Object getRppChange(Integer nbRows) {
353                try {
354                        this.setRowsPerPage(nbRows);
355                } catch (NumberFormatException nfe) {
356
357                        /**
358                         * This condition will not be reached in normal circumstances.
359                         * However, there is still keeping it as a good practise, since
360                         * input is from the gui screen.
361                         */
362                        rowsPerPage = defaultRowPerPage;
363                }
364
365                if (request.isXHR()) {
366                        this.setupDataSource();
367                        return zone.getBody();
368                } else
369                        return resources.getPage().getClass();
370
371        }
372
373        /**
374         * Event handler for inplaceupdate event triggered from nested components
375         * when an Ajax update occurs. The event context will carry the zone, which
376         * is recorded here, to allow the Grid and its sub-components to properly
377         * re-render themselves. Invokes
378         * {@link org.apache.tapestry5.services.ComponentEventResultProcessor#processResultValue(Object)}
379         * passing this (the Grid component) as the content provider for the update.
380         */
381        void onInPlaceUpdate(String zone) throws IOException {
382                // Ajax render only for events triggered by the kawwaGridPager.
383                // Events from Grid and sub-component (GridPager, ...) are handled by
384                // the Grid itself.
385                if (zone.equalsIgnoreCase(this.zone.getClientId())) {
386                        componentEventResultProcessor.processResultValue(this);
387                }
388        }

442        /**
443         * Verify if the filter form must be displayed.
444         * 
445         * @return boolean
446         */
447        public boolean isFilter() {
448                boolean result;
449                boolean bound = resources.isBound("criterium")
450                                && this.criterium != null;
451
452                if (bound && formSupport == null) {
453                        result = true;
454
455                } else {
456                        result = false;
457                }
458                return result;
459        }

469        public void onSelected(String raisedEvent) {
470                /**
471                 * Raising events have unique id as well so that there is a different
472                 * catcher that catches event for more than 1 usage of this components
473                 */
474
475                if ("filter".equalsIgnoreCase(raisedEvent)) {
476                        resources.triggerEvent(KawwaEventsConstants.UPDATE_GRID_DATA_SOURCE,null, null);
477                } else if ("cancel".equalsIgnoreCase(raisedEvent)) {
478                        resources.triggerEvent(KawwaEventsConstants.RESET_GRID_DATA_SOURCE_FILTER, null,null);
479                }
480        }

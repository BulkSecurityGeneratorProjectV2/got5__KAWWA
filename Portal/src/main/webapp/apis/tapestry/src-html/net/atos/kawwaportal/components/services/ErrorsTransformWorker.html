001package net.atos.kawwaportal.components.services;
002
003import org.apache.tapestry5.corelib.components.Errors;
004import org.apache.tapestry5.model.MutableComponentModel;
005import org.apache.tapestry5.plastic.PlasticClass;
006import org.apache.tapestry5.services.transform.ComponentClassTransformWorker2;
007import org.apache.tapestry5.services.transform.TransformationSupport;
008
009/**
010 * Worker will override the rendering of the Errors Component by adding a kawwa/Errors mixin to the errors component
011 * If you don't want this to be applied, you can use the disableKawwaErrors parameter on your component.
012**/
013public class ErrorsTransformWorker implements ComponentClassTransformWorker2 {
014
015        public void transform(PlasticClass plasticClass, TransformationSupport support,
016                        MutableComponentModel model) {
017                
018                if(model.getComponentClassName().equals(Errors.class.getName()))
019                        model.addMixinClassName(net.atos.kawwaportal.components.mixins.Errors.class.getName());
020        }
021}

001package net.atos.kawwaportal.components.services.breadcrumb;
002
003import java.util.List;
004
005import org.apache.tapestry5.ComponentResources;
006
007public interface BreadcrumbListProvider {
008        /* This method is called within the breadcrumb component.
009         * Parameters :
010         * - home : the first element of the breadcrumb (generally the home page of your application), 
011         * this is the home parameter of the breadcrumb component 
012         * - resources : the componentResources service associated with the current Breadcrumb component
013         * This method has to return a  List<String[]>, with :
014         * - String[0] is the page name (to create the pagelink)
015         * - String[1] is the text to be displayed
016         */
017        List<String[]> getBreadcrumbList (String home, ComponentResources resources);
018}

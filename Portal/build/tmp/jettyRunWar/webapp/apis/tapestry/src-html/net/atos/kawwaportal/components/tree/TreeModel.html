015package net.atos.kawwaportal.components.tree;
016
017import java.util.List;
018
019import javax.swing.tree.TreeSelectionModel;
020
021import net.atos.kawwaportal.components.components.Tree;
022
023import org.apache.tapestry5.ValueEncoder;
024
025/**
026 * A model for tree-oriented data used by the {@link Tree} component. The default implemention, {@link DefaultTreeModel}
027 * uses a {@link ValueEncoder} and a {@link TreeModelAdapter} to supply the
028 * underlying information.
029 * 
030 * @param <T>
031 *            type of data in the tree
032 * @since 5.3
033 * @see TreeSelectionModel
034 */
035public interface TreeModel<T>
036{
037    /**
038     * Returns the node or nodes that are the top level of the tree.
039     */
040    List<TreeNode<T>> getRootNodes();
041
042    /**
043     * Locates a node in the tree by its unique id.
044     * 
045     * @throws IllegalArgumentException
046     *             if no such node exists
047     * @see TreeNode#getId()
048     */
049    TreeNode<T> getById(String id);
050
051    /**
052     * Recursively searches from the root nodes to find the tree node that matches
053     * the provided element.
054     * 
055     * @param element
056     *            to search for
057     * @return matching node, or null if not found
058     */
059    TreeNode<T> find(T element);
060}

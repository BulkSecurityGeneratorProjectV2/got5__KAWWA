001package net.atos.kawwaportal.components.data;
002
003import java.util.ArrayList;
004import java.util.List;
005
006
007public class DeliveryMode {
008        
009        private String ref;
010        
011        private String label;
012        
013        private List<GMapSelectLocation> DeliveryLocationsList;
014        
015        public String getRef() {
016                return ref;
017        }
018
019        public void setRef(String ref) {
020                this.ref = ref;
021        }
022
023        public String getLabel() {
024                return label;
025        }
026
027        public void setLabel(String label) {
028                this.label = label;
029        }
030
031        public List<GMapSelectLocation> getDeliveryLocationsList() {
032                return DeliveryLocationsList;
033        }
034
035        public void setDeliveryLocationsList(
036                        List<GMapSelectLocation> deliveryLocationsList) {
037                DeliveryLocationsList = deliveryLocationsList;
038        }
039        
040        public void addDeliveryLocation(GMapSelectLocation deliveryLocation) {
041                if(this.DeliveryLocationsList == null) this.DeliveryLocationsList = new ArrayList<GMapSelectLocation>(); 
042                this.DeliveryLocationsList.add(deliveryLocation);
043        }
044
045}
